video insertion method .attr() data-video ( Not ajax but can be useful )

  • Radzone
  • Madagascar
  • Big Bunny

CODE HTML 

<video id="videoclip" poster= "images/Radzone.jpg" controls width="80%" title="Video title">
    <source id="mp4video" src="video/Radzone.mp4" type="video/mp4"/>
</video>
<div class="list-item">
 <ul>
    <li class="item" data-video ="video/Radzone.mp4">
    <a href="javascript:void(0)">Radzone</a></li>
    <li class="item" data-video ="video/Madagascar.mp4">
    <a href="javascript:void(0)">Madagascar</a></li>
    <li class="item" data-video ="video/mov_bbb.mp4">
    <a href="javascript:void(0)">Big Bunny</a></li>
  </ul>
</div>

CODE JQUERY 

   $(document).ready(function() {
     $(".list-item").find(".item").on("click", function() {
        videoData = $(this).data("video");
        videoSource = $("#videoclip").find("#mp4video");
        videoSource.attr("src", videoData);
        autoplayVideo = $("#videoclip").get(0);
        autoplayVideo.load();
        autoplayVideo.play();
    });
    });

      <thead>

          <th>Description</th>
        </tr>
      </thead>
      <tbody id="tableBody">

      </tbody>

    </table>

CODE JQUERY 

$(document).ready(function() {
 $('#btnxml').click(function () {
  $('#tableBody').empty();  // Clear the table body.
  $.ajax({
    type: 'GET',
    url: 'files/book.xml',    // The file path.
    dataType: 'xml',
    success: function(xml) {
     $(xml).find('book').each(function() {
      $('#tableBody').append(
        '<tr>' +
        '<td>' +
        $(this).find('author').text() + '</td> ' +
        '<td>' +
        $(this).find('title').text() + '</td> ' +
        '<td>' +
        $(this).find('genre').text() + '</td> ' +
        '<td>' +
        $(this).find('price').text() + '</td> ' +
        '<td>' +
        $(this).find('description').text() + '</td> ' +
        '</tr>');
       });
        }
     });
  });
});

 var output =
 "<table id=table_jason><tbody><thead><tr><th>Author</th><th>Price</th><th>genre</th><th>title</th><th>description</th></thead>";
      for (var i in result) {
        output +=

        "</td><td>" +
