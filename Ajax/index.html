video insertion method .attr() data-video ( Not ajax but can be useful )

  • Radzone
  • Madagascar
  • Big Bunny

CODE HTML

<video id="videoclip" poster= "images/Radzone.jpg" controls="controls" title="Video title">
    <source id="mp4video" src="video/Radzone.mp4" type="video/mp4"/>
</video>
<div class="list-item">
 <ul>
    <li class="item" data-video ="video/Radzone.mp4"><a href="javascript:void(0)">Radzone</a></li>
   <li class="item" data-video ="video/Madagascar.mp4"><a href="javascript:void(0)">Madagascar</a></li>
     <li class="item" data-video ="video/mov_bbb.mp4"><a href="javascript:void(0)">Big Bunny</a></li>
  </ul>
</div>

CODE JQUERY

   $(document).ready(function() {
     $(".list-item").find(".item").on("click", function() {
        videoData = $(this).data("video");
        videoSource = $("#videoclip").find("#mp4video");
        videoSource.attr("src", videoData);
        autoplayVideo = $("#videoclip").get(0);
        autoplayVideo.load();
        autoplayVideo.play();
    });
    });

AJAX jQuery

AJAX method .ajax()

CODE HTML

      <button class="appelvideo"> Get Video</button>
      <button class="appelimg"> Get img</button>
      <button class="appeltable"> Get table</button>

    <div id="objet1"> </div>

CODE JQUERY

      $(document).ready(function() {
       $("button.appelvideo").click(function() {
          $.ajax({
            url: "files/video.html",
            success: function(result) {
              $("#objet1").html(result);
            }
          });
        });
        $("button.appelimg").click(function() {
          $.ajax({
            url: "files/img.html",
            success: function(result) {
              $("#objet1").html(result);
            }
          });
        });
        $("button.appeltable").click(function() {
          $.ajax({
            url: "files/XLS.html",
            success: function(result) {
              $("#objet1").html(result);
            }
          });
        });
        });
